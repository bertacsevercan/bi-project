import pandas as pd
import matplotlib.pyplot as plt

customer_data = pd.read_csv('food_order.csv')

# Create a new DataFrame with customer segments based on unique customer IDs
uniq_customer_count = customer_data['customer_id'].value_counts()

# Define a function to categorize the customer segments
def categorize_segments(count):
    if count >= 5:
        return 'Regulars'
    elif count >= 2:
        return 'Occasional Diners'
    else:
        return 'First-time Visitors'

# Apply the categorization function to create the 'customer_segment' column
customer_data['customer_segment'] = customer_data['customer_id'].map(uniq_customer_count).apply(categorize_segments)

# Manually specify the order of categories
category_order = ['First-time Visitors', 'Occasional Diners', 'Regulars']

# Convert 'customer_segment' to a categorical variable with specified order
customer_data['customer_segment'] = pd.Categorical(customer_data['customer_segment'], categories=category_order, ordered=True)

# Plot a bar chart using matplotlib
plt.figure(figsize=(8, 6))
ax = customer_data['customer_segment'].value_counts().sort_index().plot(kind='bar', color=['red', 'green', 'blue'])
plt.title('Distribution of Customers Across Segments')
plt.xlabel('Customer Segments')
plt.ylabel('Number of Customers')

# Annotate each bar with the exact count
for p in ax.patches:
    ax.annotate(str(p.get_height()), (p.get_x() + p.get_width() / 2., p.get_height()),
                ha='center', va='center', xytext=(0, 10), textcoords='offset points')
    
plt.show()

import pandas as pd
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression


customer_data = pd.read_csv('food_order.csv')

# Show the number of ratings equal to "Not given"
print("Number of Ratings Equal to 'Not given':")
print(customer_data[customer_data['rating'] == 'Not given'].shape[0]) # 736

# Show the percentage of ratings equal to "Not given"
print("Percentage of Ratings Equal to 'Not given':")
print(customer_data[customer_data['rating'] == 'Not given'].shape[0] / customer_data['rating'].shape[0] * 100)
# 39%

# Exclude entries with "Not given" ratings
customer_data = customer_data[customer_data['rating'] != 'Not given']

# Convert 'rating' column to numeric, handling errors by setting 'Not given' to NaN
customer_data['rating'] = pd.to_numeric(customer_data['rating'], errors='coerce')

# Calculate the sum of food preparation time and delivery time
customer_data['total_time'] = customer_data['food_preparation_time'] + customer_data['delivery_time']

# Plot a scatter plot with a trendline
plt.figure(figsize=(10, 8))
plt.scatter(customer_data['total_time'], customer_data['rating'], color='blue', alpha=0.5, label='Data points')

# Fit a linear regression model
model = LinearRegression()
X = customer_data['total_time'].values.reshape(-1, 1)
y = customer_data['rating'].values.reshape(-1, 1)
model.fit(X, y)

# Plot the trendline
plt.plot(X, model.predict(X), color='red', linewidth=2, label='Trendline')

plt.title('Correlation Between Total Time and Rating')
plt.xlabel('Total Time (minutes)')
plt.ylabel('Rating')
plt.legend()
plt.grid(True)
plt.show()
